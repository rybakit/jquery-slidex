{% highlight js %} $('.slideshow').slideshow({ interval: 2, transition: function(from, to, callback) { var effect = [ 'clip', 'drop', 'explode', 'fold', 'puff' ][Math.floor(Math.random() * 5)]; to.addClass('next'); from.effect(effect, {}, this.config.speed, function() { to.addClass('current').removeClass('next'); from.removeClass('current').show(); callback.call(this); }); }}); {% endhighlight %}
